## 6. 재현 방법

```bash
# 1. Docker 환경 구성
cd /neunexus/gopedia
cp .env.local.example .env
# .env에 POSTGRES_PASSWORD 등 설정

# 2. 임베딩 서비스 + 전체 스택 시작
sudo docker compose -f docker-compose.dev.yml --env-file .env up -d postgres_db qdrant typedb
sudo docker exec -i gopedia-local-postgres_db-1 \
  psql -U admin_gopedia -d gopedia < core/ontology_so/postgres_ddl.sql
sudo docker compose -f docker-compose.dev.yml --env-file .env --profile app up -d

# 3. 임베딩 서비스 확인 (multilingual-e5-large, 1024 dims)
curl -s http://127.0.0.1:18789/health

# 4. Gopedia 확인 (embedding backend: local, vector_size=1024)
sudo docker logs gopedia-local-gopedia-1 | grep -E "embedding|vector_size"

# 5. 인제스트
for proj in gopedia neunexus osteon taxon; do
  GOPEDIA_PHLOEM_GRPC_ADDR=localhost:50051 \
    .venv/bin/python -m property.root_props.run /neunexus/geneso/universitas/$proj
done

# 6. 수동 검색 품질 확인
# 로컬 Python에서 직접 임베딩 서비스 사용
LOCAL_EMBEDDING_ADDR=http://127.0.0.1:18789 \
  .venv/bin/python -c "
from flows.xylem_flow.retriever import embed_query_local, qdrant_search_l3_points
q = 'gopedia smart sink routing strategy'
vec = embed_query_local(q, 'http://127.0.0.1:18789')
hits = qdrant_search_l3_points(vec, '127.0.0.1', 6333, 'gopedia_markdown', limit=1)
print(f'score={hits[0].score:.3f}')
"

# 7. gardener_gopedia IR 지표 측정
cd /neunexus/gardener_gopedia
.venv/bin/python -c "from gardener_gopedia.db_bootstrap import run_init_db; run_init_db()"
nohup .venv/bin/uvicorn gardener_gopedia.main:app --host 0.0.0.0 --port 18880 &
DATASET_ID=$(curl -s -X POST http://127.0.0.1:18880/datasets \
  -H "Content-Type: application/json" \
  -d @dataset/universitas_eval_bronze.json | python3 -c "import sys,json; print(json.load(sys.stdin)['id'])")
curl -s -X POST "http://127.0.0.1:18880/datasets/${DATASET_ID}/resolve-qrels?force=true"
RUN_ID=$(curl -s -X POST http://127.0.0.1:18880/runs \
  -H "Content-Type: application/json" \
  -d "{\"dataset_id\":\"${DATASET_ID}\",\"top_k\":10,\"index_version\":\"v0.3.0\"}" \
  | python3 -c "import sys,json; print(json.load(sys.stdin)['id'])")
sleep 90
curl -s "http://127.0.0.1:18880/runs/${RUN_ID}/metrics" | python3 -c "
import sys,json
for m in json.load(sys.stdin):
    if m['scope']=='aggregate' and not m['metric_name'].startswith('summary'):
        print(f'{m[\"metric_name\"]}: {m[\"value\"]:.4f}')
"
```
