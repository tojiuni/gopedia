## 5. 재현 방법

```bash
# 1. Ubuntu 24.04에 Docker 설치
sudo apt-get update && sudo apt-get install -y docker-ce docker-ce-cli containerd.io \
  docker-buildx-plugin docker-compose-plugin

# 2. 스택 시작
cd /neunexus/gopedia
cp .env.local.example .env
# .env에 OPENAI_API_KEY 설정

sudo docker compose -f docker-compose.dev.yml --env-file .env up -d postgres_db qdrant typedb
sudo docker exec -i gopedia-local-postgres_db-1 \
  psql -U admin_gopedia -d gopedia < core/ontology_so/postgres_ddl.sql

# 3. gopedia 앱(Phloem) 빌드 및 시작
sudo docker compose -f docker-compose.dev.yml --env-file .env --profile app up -d

# 4. 인제스트 (neunexus + gopedia 벡터 품질 테스트 대상)
python3 -m venv .venv && .venv/bin/pip install -r requirements.txt -q
GOPEDIA_PHLOEM_GRPC_ADDR=localhost:50051 \
  .venv/bin/python -m property.root_props.run /neunexus/geneso/universitas/neunexus
GOPEDIA_PHLOEM_GRPC_ADDR=localhost:50051 \
  .venv/bin/python -m property.root_props.run /neunexus/geneso/universitas/gopedia

# 4b. 전체 Universitas 인제스트 (gardener IR 지표 대상)
for proj in architecture gopedia neunexus osteon taxon; do
  GOPEDIA_PHLOEM_GRPC_ADDR=localhost:50051 \
    .venv/bin/python -m property.root_props.run /neunexus/geneso/universitas/$proj
done

# 5. 수동 검색 품질 확인
curl -s "http://127.0.0.1:18787/api/search?q=traefik+port+80+443+forwarding&format=json&limit=1"
curl -s "http://127.0.0.1:18787/api/search?q=gopedia+smart+sink+routing+strategy&format=json&limit=1"

# 6. gardener_gopedia IR 지표 측정
cd /neunexus/gardener_gopedia
python3 -m venv .venv && .venv/bin/pip install -r requirements.txt -q
# gardener_eval 스키마 초기화
.venv/bin/python -c "from gardener_gopedia.db_bootstrap import init_db; init_db()"

# API 기동 (백그라운드)
.venv/bin/uvicorn gardener_gopedia.main:app --host 0.0.0.0 --port 18880 &

# 데이터셋 등록 및 qrel 해석
```
